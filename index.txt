  <div class="container mt-5">
    <h1>Image Metadata Comparison</h1>
    <form id="imageForm" enctype="multipart/form-data">
      <div class="form-group file-input">
        <label for="imageInput">Select Images (Multiple)</label>
        <input type="file" class="form-control-file" id="imageInput" name="images[]" accept="image/*" multiple required>
      </div>
      <button type="button" class="btn btn-primary btn-block" onclick="compareImages()">Compare Images</button>
    </form>

    <div id="imageGrid" style="display: none;"></div>

    <div id="comparisonResults" style="display: none;">
      <h3>Metadata Comparison</h3>
      <table class="table table-bordered">
        <thead>
          <tr>
            <th>Parameter</th>
          </tr>
        </thead>
        <tbody id="comparisonTableBody"></tbody>
      </table>
    </div>
  </div>

  <script>
    function compareImages() {
      const formData = new FormData(document.getElementById('imageForm'));

      fetch('/compare', {
        method: 'POST',
        body: formData,
      })
      .then(response => {
        if (!response.ok) {
          throw new Error('Network response was not ok');
        }
        return response.json();
      })
      .then(data => {
        handleComparisonData(data);
      })
      .catch(error => {
        console.error('There was a problem with the fetch operation:', error);
      });
    }

    function handleComparisonData(data) {
      if (data && data.comparisonData) {
        const comparisonData = data.comparisonData;
        const comparisonTableBody = document.getElementById('comparisonTableBody');
        comparisonTableBody.innerHTML = '';

        const tableHeaders = comparisonData[0].imageValues.map((_, index) => `<th>Image ${index + 1} Value</th>`).join('');
        document.querySelector('thead tr').insertAdjacentHTML('beforeend', tableHeaders);

        comparisonData.forEach(rowData => {
          const row = document.createElement('tr');
          row.innerHTML = `
            <td>${rowData.parameter}</td>
            ${rowData.imageValues.map(value => `<td>${value}</td>`).join('')}
          `;
          comparisonTableBody.appendChild(row);
        });

        document.getElementById('imageGrid').style.display = 'block';
        document.getElementById('comparisonResults').style.display = 'block';
      } else {
        console.error('No comparison data received.');
      }
    }
  </script>
